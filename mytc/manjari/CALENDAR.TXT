int daysinmon[12] = { 31,28,31,30,31,30,31,31,30,31,30,31 };
class date
{	int dd,mm;
	public:
		date(int nd, int nm)
		{	dd = nd; mm = nm;	}
		int dttodtnum();
		date dtnumtodt(int);
		date futuredate(int);
		void showdate(date);
};

int date::dttodtnum()
{	int daynum = 0;
	for (int i = 1; i< mm; i++)
		daynum = daynum + daysinmon[i];
	return (daynum + dd);
}

date date::dtnumtodt(int dn)
{	date temp(0,0); int i = 1;
	int d = daysinmon[i];
	for ( ; ; )
	{	if ( (d + daysinmon[i+1]) > dn ) break;
		d += daysinmon[i++];
	}
	temp.mm = i;
	temp.dd = dn - d;
	cout<<"\n"<< dd <<" : "<< mm;
	return temp;
}

{	date temp2(0,0);

	return temp2;
}
