SELECT ENAME,JOB,HIREDATE
FROM (SELECT ENAME,JOB,HIREDATE,TO_CHAR(SYSDATE,'J')-TO_CHAR(HIREDATE,'J') DAYS1,TO_CHAR(SYSDATE,'J')-TO_CHAR(TO_DATE('20-02-1981','DD-MM-YYYY'),'J') DAYS2,TO_CHAR(SYSDATE,'J')-TO_CHAR(TO_DATE('01-05-1981','DD-MM-YYYY'),'J') DAYS3
FROM EMP)
WHERE DAYS1<DAYS2 AND DAYS1>DAYS3
ORDER BY DAYS1 DESC;

SELECT ENAME,DEPT.DNAME
FROM EMP,DEPT
WHERE EMP.DEPTNO=DEPT.DEPTNO
ORDER BY DEPT.DEPTNO;

OR

SELECT ENAME,D.DNAME
FROM 
(
SELECT DNAME,DEPTNO
FROM DEPT
WHERE DNAME='&DEPARTMENT_NAME'
) D,
(
SELECT ENAME,DEPTNO
FROM EMP
ORDER BY DEPTNO
) E
WHERE E.DEPTNO = D.DEPTNO;
------------------------------------------------------------------
SELECT ENAME,MGR,EM.SAL
FROM
(
SELECT ENAME,SAL,MGR,ROW_NUMBER()OVER(ORDER BY SAL) AS R
FROM
(
SELECT EMPNO
FROM EMP
WHERE ENAME='&MANAGER_NAME'
) E,EMP
WHERE E.EMPNO=MGR
) EM
WHERE R = 1;
------------------------------------------------------------------
SELECT UNIQUE E1.EMPNO,E1.ENAME,E1.SAL
FROM
(
SELECT EMPNO,ENAME,SAL,DEPTNO,D.A
FROM
(
SELECT AVG(SAL) A
FROM EMP
) D,EMP
WHERE SAL > D.A
) E1,
(
SELECT DEPTNO
FROM EMP 
WHERE ENAME LIKE '%T%'
) E2
WHERE E1.DEPTNO = E2.DEPTNO;
------------------------------------------------------------------
SELECT E.ENAME,E.SAL,A.AV
FROM
(
	SELECT AVG(SAL) AV
	FROM
	(
		SELECT ENAME,SAL
		FROM
		(
			SELECT DEPTNO,AVG(SAL) A
			FROM EMP
			GROUP BY DEPTNO
		) D,EMP
		WHERE D.DEPTNO=EMP.DEPTNO AND SAL > D.A
	)
) A,
(
	SELECT ENAME,SAL
	FROM
	(
		SELECT DEPTNO,AVG(SAL) A
		FROM EMP
		GROUP BY DEPTNO
	) D,EMP
	WHERE D.DEPTNO=EMP.DEPTNO AND SAL > D.A
) E
;

OR

SELECT ENAME,SAL,D.A
FROM
(
SELECT DEPTNO,AVG(SAL) A
FROM EMP
GROUP BY DEPTNO
) D,EMP
WHERE D.DEPTNO=EMP.DEPTNO AND SAL > D.A;
