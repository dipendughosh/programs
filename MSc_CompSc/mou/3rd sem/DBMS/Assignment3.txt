SELECT EMP1.ENAME AS EMP_NAME,EMP2.ENAME AS MAN_NAME
FROM EMP EMP1,EMP EMP2
WHERE EMP1.MGR=EMP2.EMPNO;

2
SELECT *
FROM
(SELECT ROW_NUMBER()OVER(ORDER BY EMPNO) AS R_NO,EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO
FROM EMP)EM
WHERE R_NO = ANY(1,3,5,7,9);

SELECT *
FROM
(SELECT ROW_NUMBER()OVER(ORDER BY SAL DESC) AS R_NO,EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO
FROM EMP)EM
WHERE R_NO = 5;

SELECT EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO,COUNT(*)
FROM EMP
GROUP BY EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO
HAVING COUNT(*) > 1;

11
SELECT R_NO,EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO 
FROM
(SELECT MOD(EM.R_NO,2) AS MD,R_NO,EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO
FROM
(SELECT ROW_NUMBER()OVER(ORDER BY EMPNO) AS R_NO,EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO
FROM EMP)EM)
WHERE MD = 0;

SELECT R_NO,EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO 
FROM
(SELECT MOD(EM.R_NO,2) AS MD,R_NO,EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO
FROM
(SELECT ROW_NUMBER()OVER(ORDER BY EMPNO) AS R_NO,EMPNO,ENAME,JOB,MGR,HIREDATE,SAL,COMM,DEPTNO
FROM EMP)EM)
WHERE MD != 0;

SELECT ENAME,TO_CHAR(TO_DATE('01-03-1990','DD-MM-YYYY'),'J')-TO_CHAR(HIREDATE,'J') DAYS
FROM EMP
WHERE TO_CHAR(TO_DATE('01-03-1990','DD-MM-YYYY'),'J')-TO_CHAR(HIREDATE,'J') > 365*5;
