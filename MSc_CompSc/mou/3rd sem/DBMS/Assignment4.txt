SELECT SUBSTR(ENAME,2,4) AS NAME
FROM EMP;

SELECT ENAME,SAL,DATETIME
FROM (SELECT CURRENT_TIMESTAMP AS DATETIME,ENAME,SAL
FROM EMP);

SELECT date_trunc('hour', timestamP) FROM DUAL;
SELECT EXTRACT(SECOND FROM TIMESTAMP CURRENT_TIMESTAMP) FROM DUAL;

SELECT ENAME,LengthOfNAME
FROM (SELECT ROW_NUMBER() OVER(ORDER BY LengthOfNAME DESC) AS R_NO,LengthOfNAME,ENAME
FROM (SELECT LENGTH(ENAME) as LengthOfNAME,ENAME
FROM EMP))
WHERE R_NO=1;

SELECT ENAME,ROUND(DAYS/365) EXPERIENCE_IN_YEARS
FROM (SELECT ENAME,TO_CHAR(SYSDATE,'J')-TO_CHAR(HIREDATE,'J') DAYS
FROM EMP);
